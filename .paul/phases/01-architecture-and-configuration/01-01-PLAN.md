## Goal
Build the **contracts and configuration scaffolding** for Phase 1 — a `providers.py` module declaring a `Provider` protocol + capability registry + per-provider config (voices, supported model patterns, output format, concurrency default, HF model revision pins where applicable), plus settings.py additions for HF cache location and model-revision constants. This plan delivers the **interfaces** that Phase 2+ will plug implementations into; no synthesis logic, no `tts_conversion.py` refactor.

## Purpose
PRD §FR-1A mandates a multi-provider architecture. The Phase 0 decisions narrowed v0.1 scope to three providers (OpenAI hosted, Ollama local, Kokoro local; VibeVoice deferred to v0.2). Wiring `tts_conversion.py` to dispatch through providers is Phase 2 work and would conflict with the user's pre-session dirty edits to that file. This plan instead lands the contract layer in a NEW module (`providers.py`) so Phase 2 has a clean target to hit, and folds in two pieces of audit-debt explicitly assigned to Phase 1: (a) cleaning up the `sys.path.insert` hack from `tests/conftest.py` (audit D3 from 00-02) and (b) recording the §14.2(4) HF cache decision in `settings.py` as code.

The regression net (101 tests, 0.77s) stays green throughout.

    Provider capability registry for Text2AudioBook (v0.1 milestone).

    VibeVoice intentionally omitted from the v0.1 registry per Phase 0 §14.2(1)
    decision (no GPU assumed in v0.1 target machines). v0.2 Phase 6.3 will add

    This module is the SINGLE SOURCE OF TRUTH for provider capabilities.
    settings.py is a thin facade and reads from here.
    """
    import re
    from dataclasses import dataclass
    from types import MappingProxyType
    from typing import Literal, Mapping


    MILESTONE = "v0.1"   # audit-added S1 — surface the milestone constraint in code, not just docs


    @dataclass(frozen=True)
    class ProviderCapability:
        name: str
        kind: Literal["hosted", "local-api", "local-hf"]
        voices: tuple[str, ...]
        model_pattern: str                          # regex; compiled at module-import below (audit M1)
        fallback_models: tuple[str, ...]
        output_format: Literal["mp3", "wav"]
        default_max_concurrency: int
        hf_model_repo: str | None = None            # e.g. "hexgrad/Kokoro-82M"; None if not HF-hosted (audit M2 — settings.py derives HF_MODEL_REVISIONS from this + hf_model_revision below)
        hf_model_revision: str | None = None        # SHA / semver tag / "main" with TODO
        is_available_in_v01: bool = True


    _ALLOWED_REVISION_LITERALS = {"main", "master"}
    _SHA_RE = re.compile(r"[0-9a-f]{40,64}")
    _SEMVER_RE = re.compile(r"v?\d+\.\d+\.\d+([+-][\w.]+)?")


    def _validate_revision(rev: str | None, *, repo: str) -> None:
        """Fail-fast at module import (audit S2). Caller MUST pass repo for error messages."""
        if rev is None:
            return
        if _SHA_RE.fullmatch(rev):
            return
        if _SEMVER_RE.fullmatch(rev):
            return
        if rev in _ALLOWED_REVISION_LITERALS:
            return
        raise ValueError(
            f"HF model revision for {repo!r} does not match accepted schema "
            f"(SHA / semver / 'main'|'master'): {rev!r}"
        )


    _RAW_REGISTRY: dict[str, ProviderCapability] = {
        "OpenAI": ProviderCapability(
            name="OpenAI",
            kind="hosted",
            voices=("alloy", "echo", "fable", "onyx", "nova", "shimmer"),
            model_pattern=r"^tts-[a-z0-9.-]+$",
            fallback_models=("tts-1", "tts-1-hd"),
            output_format="mp3",
            default_max_concurrency=2,
            hf_model_repo=None,
            hf_model_revision=None,
        ),
        "Ollama": ProviderCapability(
            name="Ollama",
            kind="local-api",
            voices=(),
            model_pattern=r".*(bark|kokoro|tts|speech).*",
            fallback_models=(),
            output_format="mp3",
            default_max_concurrency=1,
            hf_model_repo=None,
            hf_model_revision=None,
        ),
        "Kokoro": ProviderCapability(
            name="Kokoro",
            kind="local-hf",
            voices=(
                "af_heart", "af_alloy", "af_aoede", "af_bella", "af_jessica",
                "af_kore", "af_nicole", "af_nova", "af_river", "af_sarah", "af_sky",
                "am_adam", "am_echo", "am_eric", "am_fenrir", "am_liam", "am_michael",
                "am_onyx", "am_puck", "am_santa",
            ),
            model_pattern=r"^kokoro(-[a-z0-9-]+)?$",
            fallback_models=("kokoro-82m",),
            output_format="wav",
            default_max_concurrency=1,
            hf_model_repo="hexgrad/Kokoro-82M",
            hf_model_revision="496dba118d1a58f5f3db2efc88dbdc216e0483fc89fe6e47ee1f2c53f18ad1e4",
        ),
    }


    # Fail-fast validation at module import (audit M1 + S2)
    for _cap in _RAW_REGISTRY.values():
        try:
            re.compile(_cap.model_pattern)
        except re.error as exc:
            raise ValueError(
                f"Invalid model_pattern regex for provider {_cap.name!r}: {exc}"
            ) from exc
        _validate_revision(_cap.hf_model_revision, repo=_cap.hf_model_repo or _cap.name)


    # Immutable registry (audit M3)
    PROVIDER_REGISTRY: Mapping[str, ProviderCapability] = MappingProxyType(_RAW_REGISTRY)


    def get_provider_capability(name):
        if not isinstance(name, str) or not name:
            return None
        return PROVIDER_REGISTRY.get(name)


    def list_providers():
        return tuple(PROVIDER_REGISTRY.keys())
    ```

    def get_provider_capability(name):
        """Thin facade over providers.get_provider_capability — keep settings.py as the public config surface."""
        import providers

    def _hf_model_revisions():
        """Derive HF revision map from the provider registry (audit M2 — single source of truth)."""
        import providers
        return {
            cap.hf_model_repo: cap.hf_model_revision
            for cap in providers.PROVIDER_REGISTRY.values()
            if cap.hf_model_repo and cap.hf_model_revision
        }


    # Module-level dict-like view: lazy-evaluated each access (avoids stale cache after future registry edits)
    class _HFModelRevisionsView:
        def __getitem__(self, key):
            return _hf_model_revisions()[key]
        def get(self, key, default=None):
            return _hf_model_revisions().get(key, default)
        def __contains__(self, key):
            return key in _hf_model_revisions()
        def __iter__(self):
            return iter(_hf_model_revisions())
        def keys(self):
            return _hf_model_revisions().keys()
        def values(self):
            return _hf_model_revisions().values()
        def items(self):
            return _hf_model_revisions().items()
        def __len__(self):
            return len(_hf_model_revisions())
        def __repr__(self):
            return f"_HFModelRevisionsView({_hf_model_revisions()!r})"


    HF_MODEL_REVISIONS = _HFModelRevisionsView()

The view object exposes the dict-like surface but **defers** to `providers.PROVIDER_REGISTRY` on every access, so there is exactly one source of truth. If a future plan changes the Kokoro revision in `providers.py`, `settings.HF_MODEL_REVISIONS` updates automatically without code edits in `settings.py`. No duplication.
